file(GLOB_RECURSE TEXTURE_FILES CONFIGURE_DEPENDS "*.png" "*.jpg")
foreach(FILE ${TEXTURE_FILES})
    file(RELATIVE_PATH FILE_PATH ${CMAKE_CURRENT_SOURCE_DIR} ${FILE})
    get_filename_component(NEW_FILE_PATH ${CMAKE_CURRENT_BINARY_DIR}/${FILE_PATH} DIRECTORY)
    file(MAKE_DIRECTORY ${NEW_FILE_PATH})
    configure_file(${FILE} ${NEW_FILE_PATH} COPYONLY)
endforeach()
